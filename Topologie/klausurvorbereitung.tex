\subsection*{Topologische Räume}
\begin{itemize}
	\item \tbf{Metriken} sind positiv definit, symmetrisch, und erfüllen die Dreiecksungleichung.
	\item \tbf{Topologien} sind gegen endliche Schnitte und beliebige Vereinigungen abgeschlossen.
	\item \tbf{Basis} = Jede offene Menge ist eine Vereinigung von Basismengen. \tbf{S}ystem ist Basis einer Topologie, wenn endliche Schnitte erzeugt werden. \tbf{A}lternativ: Jeder Punkt aus endlichem Schnitt hat Basisumgebung im Schnitt.
	\item \tbf{Subbasis} = Offene Mengen sind genau die Vereinigungen von Schnitten von Subbasismengen. \tbf{J}ede Teilmenge ist Subbasis einer Topologie.
	\item \tbf{Umgebungsbasis} = Jede Umgebung enthält Basisumgebung. \tbf{Erstes Abz.axiom} = Abzählbare Umgebungsbasis um jeden Punkt. \tbf{Zweites Abz.axiom} = Abzählbare Basis.
	\item \tbf{Inneres} = Vereinigung offener Teilmengen. \tbf{Abschluss} = Alle Punkte, die keine zur Menge disjunkte Umgebung haben. \tbf{Rand} = Abschluss$\setminus$ Inneres. \tbf{Dicht} = Raum liegt im Abschluss einer Teilmenge. \tbf{Nirgends Dicht} = Inneres des Abschlusses ist leer. \tbf{Cantormenge} ist abgeschlossen, nirgends dicht, überabzählbar.
	\item \tbf{Spurtopologie} = Schnitte von offenen Mengen mit Teilmenge. \tbf{G}röbste Topologie, auf der Inklusion stetig ist. \tbf{A}bbildung $g$ auf Teilmenge ist stetig gdw. $\iota \circ g$ stetig. \tbf{Einbettung} = Homöomorphismus auf Bild.
	\item \tbf{Initialtopologie} = Urbilder von offenen Mengen als Subbasis, evtl. mehrere Abbildungen $f_i$. Gröbste Topologie, auf der alle $f_i$ stetig sind. Im Allgemeinen $f_i$ nicht offen. $g : Z \to Y$ stetig gdw. alle $f_i \circ g$ stetig.
	\tbf{Produkttopologie} = Initialtopologie der $p_i$. $p_i$ sind offen.
	\item \tbf{Finaltopologie} = Mengen, deren Urbilder offen sind, als Subbasis. Feinste Topologie, auf der alle $f_i$ stetig sind. $g : Y \to Z$ stetig gdw. alle $g \circ f_i$ stetig. \tbf{Quotiententopologie} = Finaltopologie der Quotientenabbildung.
	\item \tbf{Summenraum} = $Z$ offen $\Leftrightarrow$ Schnitt mit allen $X_i$ offen.
\end{itemize}
\subsection*{Stetige Abbildungen}
\begin{itemize}
	\item $f$ \tbf{stetig} $\Leftrightarrow$ Urbilder von Subbasismengen sind in offenen Mengen enthalten $\Leftrightarrow$ Einschränkungen auf abgeschlossene Überdeckungsmengen stetig.
\end{itemize}
\subsection*{Zusammenhang}
\begin{itemize}
	\item Teilmenge zusammenhängend, wenn zusammenhängend in der Spurtopologie. \tbf{Abschluss} zusammenhängender Mengen ist zsh. \tbf{Bild} zusammenhängender Mengen unter stetiger Abbildung ist zsh (ZWS). \tbf{Vereinigung} zusammenhängender Mengen mit nichtleerem Schnitt ist zsh. \tbf{Produkt} zsh $\Leftrightarrow$ Faktoren zsh.
	\item Zusammenhängend $\Leftrightarrow$ offene Überdeckung hat für je zwei Punkte \tbf{Kette} an offenen Mengen, die diese verbinden. 
	\item \tbf{Zusammenhangskomponente} = Vereinigung aller zsh. Obermengen. \tbf{Total unzusammenhängend} = alle Zshkomponenten einelementig.
	\item 
	\tbf{Lokal (weg-)zusammenhängend} = Jede Umgebung erhält (weg)-zsh. Umgebung. 
	\item (Lokal) wegzsh $\implies$ (lokal) zsh. 
	\item Lokal (weg)zsh $\implies$ \tbf{alle (Weg)zshkomponenten offen.}
	\item Zsh + lokal wegzsh $\implies$ wegzsh.
	\item \tbf{Produkt} lokal zsh gdw. Jeder Faktor lokal zsh und fast alle Faktoren zsh.
	\item Wzsh $\not \rightarrow$ lokal wzsh, wzsh $\not \rightarrow$ lokal zsh (Topologischer Sinus mit Enden verbunden)
	\item \tbf{Stetige Funktionen} bilden (weg)-zsh. Räume auf (weg)-zsh. Räume ab.
	\item Ist $X$ (weg)zsh, so auch $X/\sim$.
\end{itemize}
\subsection*{Filter}
\begin{itemize}
	\item Folgenkonvergenz reicht, falls abzählbare Umgebungsbasis existiert.
	\item \tbf{Filter} = $X$, Obermengen, Schnitte enthalten.
	\item \tbf{Filterbasis} = Filter besteht aus Obermengen.
	\item \tbf{Frei} = Schnitt aller Mengen ist leer. Sonst \tbf{fixiert}. \tbf{Fréchet-Filter} = Kofinit auf $\bN$ ist frei.
	\item \tbf{Ultrafilter} = Für jedes $A$ $A$ oder $X \setminus A$ enthalten. Jeder Filter kann zu einem Ultrafilter erweitert werden.
	\item \tbf{Filterkonvergenz} gegen $x$  = Alle Umgebungen von $x$ sind enthalten. 
	\item \tbf{Berührungspunkt} = Alle Umgebungen haben nichtleeren Schnitt mit einer Filtermenge = Es existiert ein feinerer Filter, der gegen den Punkt konvergiert. \tbf{Abschluss} = Menge aller Berührpunkte.
\end{itemize}
\subsection*{Trennungseigenschaften}
\begin{itemize}
	\item \tbf{T0} = $\cU(y) \neq \cU(x)$.
	\item \tbf{T1} = $\exists U_x, U_y$, $x \notin U_y$ und $y \notin U_x$. 
	\item \tbf{T2} = $U_x \cap U_y = \emptyset$. 
	\item \tbf{T3} = $U_x \cap U_A = \emptyset$ für $A$ abgeschlossen. 
	\item \tbf{T3a} = $\exists f : X \to [0,1]_\bR$ stetig mit $f(x) = 1, f[A] \subseteq \lr{0}$. \item \tbf{T4} = $U_A \cap U_B = \emptyset$.
	\item T1 + T3 = \tbf{regulär}. T1 + T3a = \tbf{vollständig regulär}. T1 + T4 = \tbf{normal}.
	\item T0 bis T3a vererben sich auf \tbf{beliebige Unterräume}. T4 vererbt sich auf \tbf{abgeschlossene Unterräume}.
	\item \tbf{Indiskrete Topologie} hat T3, T3a und T4, aber nicht T0. \tbf{Kofinite Topologie} auf $\bN$ hat T1, aber nicht T2.
	\item T1 $\Leftrightarrow$ \tbf{Alle Punkte abgeschlossen} $\Leftrightarrow$ Jede Menge ist Schnitt ihrer offenen Obermengen.
	\item T2 $\Leftrightarrow$ \tbf{Diagonale ist im Produkt abgeschlossen} $\Leftrightarrow$ Jeder Punkt ist Schnitt seiner geschlossenen Umgebungen $\Leftrightarrow$ jeder konvergente Filter hat genau einen Berührpunkt.
	\item Jeder vollständig reguläre Raum kann \tbf{in ein Produkt} $\prod_{f \in L}[0,1]$ eingebettet werden ($L = \lr{f \in C(X, [0,1])}$).
	\item \tbf{Produkt nichtleerer Räume} ist T0 bis T3a genau dann, wenn Faktoren T0 bis T3a sind.
	\item \tbf{Sorgenfrey} ist vollständig regulär. \tbf{Sorgenfrey} ist normal, aber \tbf{Sorgenfrey}$^2$ nicht. 
	\item \tbf{Quotiententopologie} ist genau dann T1, wenn alle Äquivalenzklassen abgeschlossen sind. 
	\item Ist Quotient Hausdorff, ist Relation im Produkt des Basisraums abgeschlossen. 
	\item Ist $\sim \subseteq X \times X$ abgeschlossen und $p$ offen, ist $X/\sim$ Hausdorff.
	\item Ist $X$ regulär und $p$ offen und abgeschlossen, ist $X/\sim$ Hausdorff.
	\item Ist $X$ T4 und $p$ abgeschlossen, ist $X/\sim$ T4.
	\item Seien $f,g : X \to Y$ stetig und $Y$ Hausdorff. Dann gilt:
	\begin{enumerate}
		\item $\lr{f(x) = g(x)}$ ist abgeschlossen.
		\item $f,g$ sind eindeutig durch ihr Bild auf dichten Teilmengen bestimmt.
		\item $\lr{f(x) = y}$ ist abgeschlossen.
		\item Ist $f$ injektiv, ist $X$ Hausdorff.
		\item Sei $x \sim y \Leftrightarrow f(x) = f(y)$. Dann ist $X/\sim$ Hausdorff.
	\end{enumerate}
	\item $X$ ist \tbf{T3} genau dann, wenn jede Umgebung eine abgeschlossene Umgebung enthält, \tbf{also wenn die geschlossenen Mengen eine Umgebungsbasis bilden}.
	\item $X$ ist \tbf{T4} genau dann, wenn zwischen jeder offenen Menge mit abgeschlossener Teilmenge eine offene Menge und ihr Abschluss liegen.
	\item Urysohn: Normal $\implies$ vollständig regulär
\end{itemize}
\subsection*{Kompaktheit}
\begin{itemize}
	\item Jede offene Überdeckung hat \tbf{endliche Teilüberdeckung}\\ $\Leftrightarrow$ Wenn ein System von abgeschlossenen Mengen \tbf{leeren Schnitt} hat, so auch ein endlicher Teilschnitt.\\ $\Leftrightarrow$ Wenn in einem System abgeschlossener Mengen der \tbf{Schnitt endlich vieler Mengen nichtleer} ist, so auch der Schnitt über das ganze System.\\ 
	$\Leftrightarrow$ Jeder Filter hat einen \tbf{Berührungspunkt}\\ $\Leftrightarrow$ Jeder \tbf{Ultrafilter} Konvergiert
	\item Ein Raum ist kompakt genau dann, wenn jede \tbf{Überdeckung mit Subbasismengen} eine endliche Teilüberdeckung hat.
	\item Es seien $X$ Hausdorff und $K \subseteq X$ kompakt. Es sei $x \in X \setminus K$. \tbf{Dann haben $x$ und $K$ disjunkte offene Umgebungen.}
	\item Jede abgeschlossene Teilmenge eines \tbf{kompakten Raums} ist kompakt.
	\item Jede kompakte Teilmenge eines \tbf{Hausdorffraums} ist abgeschlossen.
	\item Kompakte Hausdorffräume sind \tbf{T4}.
	\item Stetige Funktionen bilden kompakte Räume auf kompakte Räume ab.
	\item $f : X \to Y$ stetig, $X$ kompakt, $Y$ Hausdorff $\implies$ $f$ abgeschlossen. Ist $f$ außerdem injektiv, ist $f$ eine Einbettung.
	\item \tbf{Tychonoff:} Produkte kompakter Räume sind kompakt.
	\item \tbf{Heine-Borel:} Kompakte Teilmengen des $\bR^n$ sind genau die\\ beschränkten abgeschlossenen Mengen.
	\item Ist $X$ kompakt, so ist $X/\sim$ kompakt.
\end{itemize}

\subsection*{Topologische Gruppen}
\begin{itemize}
	\item \tbf{Gruppenoperation:} Respektiert Einheitselement, assoziiert mit Gruppenverknüpfung.
	\item Bahn $Gx$, Bahnenraum $X/G$, Standgruppe $G_A = \lr{g \in G \mid gA = A}$.
	\item Transitive Operation: $Gx = X$ für alle $x \in X$.
	\item Opertiert $G$ transitiv auf $X$, so ist die Abbildung
	\begin{align*}
		\phi_x : G/G_x &\to G\\
		[g] &\mapsto gx
	\end{align*}
	wohldefiniert und bijektiv.
	\item In einer Topologischen Gruppe sind Links- und Rechtsverknüpfungen Homöomorphismen.
	\item Sei $H \subseteq G$ Untergruppe. Dann ist die Projektion $p : G \to G/H$ stetig und offen.
	\item Ist $H$ in $G$ abschlossen und $G$ Hausdorffsch, so ist $G/H$ Hausdorffsch. Ist $G/H$ T1, ist $H$ abschlossen.
\end{itemize}
\subsection*{Homotopie}
\begin{itemize}
	\item Stetige $f,g : X \to Y$ heißen \tbf{homotop mittels $h$,}\\ wenn $h : X \times [0,1] \to Y$ stetig ist mit
	\begin{align*}
		\forall x \in X:
		h(x,0) = f(x),
		h(x,1) = g(x)
	\end{align*}
	schreibe $f \sim g$ oder $f \sim_h g$.
	\item $f,g$ heißen \tbf{homotop relativ zu $A$}, falls die Abbildungen homotop sind und alle Zwischenabbildungen auf $A$ übereinstimmen: $h(-,t)|_A = f|_A = g|_A$ für alle $t \in [0,1]$.
	\item Homotopie ist eine Äquivalenzrelation. Verknüpfungen homotoper Funktionen sind homotop ($f_1 \sim f_2, f_1 \sim g_2 \implies f_1 \circ g_1 \sim f_2 \circ g_2$)
	\item Eine stetige Abbildung $f : X \to Y$ heißt \tbf{Homotopieäquivalenz zwischen $X$ und $Y$}, falls es $g : Y \to X$ so gibt, dass $g \circ f$ und $f \circ g$ zu den jeweiligen Identitätsfunktionen homotop sind.
	\item Kompaktheit und topologische Dimension sind nicht invariant unter Homotopieäquivalenz.
	\item Sei $A \subseteq X$. Dann sind $A$ und $X$ homotopieäquivalent, falls die Identität auf $X$ homotop zu einer Funktion ist, deren Bild $A$ ist und deren Einschränkung auf $A$ die Identität ist. (Also zu einer Retraktion von $X$ auf $A$.)
	\item Ein Raum heißt \tbf{zusammenziehbar}, falls er zu einem Einpunktraum $\lr{x_0}$ homotopieäquivalent ist.
	\item $X$ ist genau dann zusammenziehbar, wenn es eine stetige Abbildung $h : X \times I \to X$ und $x_0 \in X$ so gibt, dass $h(x,0) = x$ und $h(x,1) = x_0$.
\end{itemize}
\subsection*{Schleifen und Fundamentalgruppe}
\begin{itemize}
	\item Eine Schleife ist ein Weg $\gamma : [0,1] \to X$ mit $\gamma(0) = \gamma(1)$.
	\item Die Menge der Schleifen mit gleichem Basispunkt $x_0$ Modulo Relativhomotopieauf $\lr{0,1}$ bilden mit der Verknüpfung
	\begin{align*}
		(f * g)(t) = \begin{cases}
			f(2t) & 0 \leq t \leq \frac{1}{2};\\
			g(2t-1) & \frac{1}{2} \leq t \leq 1
		\end{cases}
	\end{align*}
	die \tbf{Fundamentalgruppe}
	\begin{align*}
		\pi_1(X,x_0) = \Omega(X,x_0) / \sim_{\lr{0,1}}
	\end{align*}
	Das neutrale Element ist die konstante Schleife $\epsilon_{x_0} := (x \mapsto x_0)$.
	\item Das Inverse zu einer Schleife $\gamma$ ist $\ol \gamma(t) := \gamma(1-t)$.
	\item Ist $\gamma : [0,1] \to X$ eine Schleife und $\phi : [0,1] \to [0,1]$ stetig mit $\phi(0) = 0$ und $\phi(1) = 1$, so sind $\gamma \circ \phi$ und $\gamma$ $\lr{0,1}$-homotop.
	\item Die Abbildung $\pi_1$, die jedem topologischen Raum mit Basispunkt seine Fundamentalgruppe zuordnet, wobei wir $f : (X,x) \to (Y,y)$ auf
	\begin{align*}
		\pi_1(f) := f^* : ([\gamma] \mapsto [f \circ \gamma])
	\end{align*}
	abbilden, ist ein kovarianter Funktor, also gilt $$\pi_1(\id_X)[\gamma] = [\id_X \circ \gamma] = [\gamma] \implies \pi_1(\id_X) = \id_{\pi(X,x)}$$
	und
	\begin{align*}
		\pi_1(f) \circ &=
		\pi_1(f \circ g)[\gamma]
	\end{align*}
	Außerdem ist $f^* := \pi_1(f)$ ein Gruppenhomomorphismus, also tatsächlich ein Morphismus in der Kategorie \ttt{Top+}.
	\item Jede Homotopieklasse von Wegen von $x$ nach $y$ induziert einen Gruppenisomorphismus
	\begin{align*}
		J_\delta : \pi_1(X,x) &\to \pi_1(X,y)\\
		[\gamma] &\mapsto [\ol \delta * \gamma * \delta]
	\end{align*}
	für $\delta$ ein beliebiger Weg von $x$ nach $y$. Insbesondere sind alle Fundamentalgruppen isomorph, wenn $X$ wegzusammenhängend ist.
	\item Zwei Schleifen $\gamma_0, \gamma_1$ heißen \tbf{geschlossen homotop}, wenn es eine homotopie gibt, die zu jedem Zeitpunkt eine Schleife ist, also 
	\begin{align*}
		h(s,0) &:= h_0(s) = \gamma_0(s),\\
		h(s,1) &:= h_1(s) = \gamma_1(s),\\
		h(0,t) &:= h_t(0) = h_t(1) = h(1,t)
	\end{align*}
	für alle $s,t \in [0,1]$.
	\item Relative Homotopie = Geschlossene Homotopie, in der sich der Basispunkt nicht bewegt.
	\item Eine Schleife heißt zusammenziehbar, wenn sie geschlossen homotop zu einer konstanten Schleife $\epsilon_x$ ist.
	\item Eine Schleife $\gamma$ ist genau dann zusammenziehbar, wenn sie Relativhomotop zur konstanten Schleife an ihrem Basispunkt ist. (Durch Konjugation aller $h_t$ mit einem Weg $\gamma(0) \to x$, wie beim Gruppenisomorphismus).
	\item Eine Schleife $\gamma : [0,1] \to X$ ist genau dann Zusammenziehbar, wenn die quotientierte Abbildung $\tilde \gamma : S^1 \to X$ eine stetige Fortsetzung auf $\ol B^2$ hat, also $\exists f : \ol B^2 \to X, f|_{S^1} = \tilde \gamma$.
	\item Für jede Homotopieäquivalenz $f : X \to Y$ und alle $x_0 \in X$ ist die Abbildung $\pi_1(f) = f^* = f^*_{x_0} : \pi_1(X,x_0) \to \pi_1(Y, f(x_0))$ ein Gruppenisomorphismus.
	\item 
	Insbesondere existiert in jedem zusammenziehbaren Raum per Definition eine Homotopieäquivalenz zum Einpunktraum $\lr{x_0}$, also ist die Fundamentalgruppe trivial.
\end{itemize}

\subsection*{Retraktionen}
\begin{itemize}
	\item Es sei $X$ ein topologischer Raum und $A \subseteq X$. Dann heißt eine stetige Abbildung $r : X \to A$ eine \tbf{Retraktion}, wenn $r|_A = \id_A$.
	\item Es gibt keine Retraktion $\ol{B^2} \to S^1$.
	\item \tbf{Brouwer:} Jede stetige Funktion $f : \ol{B^2} \to \ol{B^2}$ hat einen Fixpunkt.
	\item \tbf{Borsuk-Ulam:} Sei $f : S^2 \to \bR^2$ stetig. Dann gibt es ein $x \in S^2$ mit $f(x) = f(-x)$.
\end{itemize}
\subsection*{Freies Produkt}
\begin{itemize}
	\item \tbf{Freies Produkt:} Buchstaben aus der gleichen Gruppe werden gekürzt, das neutrale Element wird ignoriert.
	\item \tbf{Seifert-van Kampen:} Es seien $U,V$ topologische Räume, $X = U \cap V$, $U,V$ offen, $x_0 \in U \cap V$, und $U,V,U \cap V$ wegzusammenhängend. Dann ist $\pi_1(X,x_0)$ ein homomorphes Bild von $\pi_(U,x_0) \star \pi_1(V,x_0)$. Falls $U \cap V$ zusammenziehbar ist, liegt ein Isomorphismus vor.
	\item \tbf{Korollar:} Ist $X = U \cup V$, $U,V$ offen, wzhgd,zzb, $U \cap V$ nichtleer und wzhgd, so ist $X$ zzb.
	\item \tbf{Korollar:} $\pi_1(S^n,x)$ ist für $n \geq 2$ für alle $x$ trivial.  
\end{itemize}
